REST Interface JavaScript (jQuery) Samples

 

Login

var form = new FormData();

form.append("size", "original");

form.append("username", "{{username}}");

form.append("password", "{{password}}");

 

var settings = {

  "async": true,

  "crossDomain": true,

  "url": "http://ggrigore-z800.na.tibco.com/dev/v1/login",                 

  "method": "POST",

  "headers": {},

  "processData": false,

  "contentType": false,

  "mimeType": "multipart/form-data",

  "data": form

}

 

$.ajax(settings).done(function (response) {

  console.log(response);

});
